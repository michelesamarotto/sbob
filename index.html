Master's thesis supplementary video documentation.

Title: Sonic Bits & Optic Bytes. Aesthetic Considerations on and Practical Explorations of Audiovisual Synthesis at Byte-Level via Data Encoding and Unix IPC
Author: Michele Samarotto
Supervisors: Prof. Dr. Christoph Seibert, Dr. Bernd Härpfer
University/Institute: University of Music Karlsruhe / Institute for Music Informatics and Musicology
Date: June 2021

For browser display, all videos have been transcoded from the original raw-formats using the NUT container to the WebM container using the VP9 and Opus codecs and a 2-pass encoding:

$ ffmpeg -i source.nut -c:v libvpx-vp9 -pass 1 -b:v 0 -crf 15 -pix_fmt yuv420p -threads 8 -speed 4 -tile-columns 6 -frame-parallel 1 -an -f webm /dev/null

$ ffmpeg -i source.nut -c:v libvpx-vp9 -pass 2 -b:v 0 -crf 15 -pix_fmt yuv420p -threads 8 -speed 2 -tile-columns 6 -frame-parallel 1 -auto-alt-ref 1 -lag-in-frames 25 -c:a libopus -ar 48000 -ac 2 -b:a 320k -f webm out.webm

See: https://sites.google.com/a/webmproject.org/wiki/ffmpeg/vp9-encoding-guide

FM (Chapter 5.1)

$ mkfifo /tmp/tee.fifo /tmp/audio.fifo /tmp/video.fifo

$ ffmpeg -f lavfi -i aevalsrc="'sin((1378.125+sin(200.0*2*PI*t))*2*PI*t)':s=44100" -f u8 -ar 44100 -ac 1 - < /dev/null 2> /dev/null | ffmpeg -f rawvideo -pixel_format rgb24 -video_size 32x32 -i - -f rawvideo -pixel_format rgb24 -video_size 32x32 - > /tmp/tee.fifo 2> /dev/null &

$ tee < /tmp/tee.fifo >(pv -pterbTCB 5M | cat - > /tmp/video.fifo) | pv -pterbTCB 5M | cat - > /tmp/audio.fifo

$ ffmpeg -f rawvideo -pixel_format rgb24 -video_size 32x32 -framerate 14.35546875 -thread_queue_size 512 -i /tmp/video.fifo -f u8 -ar 44100 -ac 1 -thread_queue_size 512 -i /tmp/audio.fifo -f nut -c:v rawvideo -pixel_format rgb24 -video_size 32x32 -framerate 14.35546875 -c:a pcm_u8 -sample_fmt u8 -ar 44100 -ac 1 -vf "scale=w=720:h=720:flags=bicubic" -t 00:01:00 fm.nut

RGB Bit-Shift (Chapter 5.6)

marble.pde 

int c;
int file_count = 0;
float freq = 64.0;
float amp = 100.0;
float x_inc = 0.01;
float y_inc = 0.01;
float pixel_size = 0.1;
byte[] bytes;
String out_dir = ”frames/frame”;

void setup() {
  bytes = new byte[32 * 32];
  size(32, 32, P3D);
  background(0);
}

void draw() {
  float x_off = 0.0;

  loadPixels();

  for (int x = 0; x < width; x++) { 
    float y_off = 0.0;

    for (int y = 0; y < height; y++) {

      c = stripe_turbulence(x, x_off, y_off, freq, pixel_size, amp);

      pixels[x + y * width] = color(c);
      bytes[x + y * width] = byte(c);

      y_off += y_inc;
    }

    x_off += x_inc;
  }

  updatePixels();
  saveBytes(out_dir + file_count, bytes);

  amp += 1.0;
  println(file_count);

  file_count += 1;
}

float turbulence(float x, float y, float pixel_size) {
  float t = 0;
  float scale = 1;

  while (scale > pixel_size) {
    t += abs(noise(x / scale, y / scale) * scale); scale /= 2;
  }

  return t;
}

int stripe_turbulence(int x, float x_off, float y_off, float freq, float pixel_size, float amp) {
  float t;

  t = (1 + cos((PI * (x + turbulence(x_off, y_off, pixel_size) * amp)) /freq)) / 2;

  return int((1 - t) * 255);
}

shifter.c 

#include <stdio.h>
#include <unistd.h>
#include <stdlib.h>
#include <math.h>

#define TRUE 1
#define FALSE 0

int main(int argc, char *argv[])
{
  int num_bits = atoi(argv[1]);
  int b_size = num_bits / 8;
  int num_shift = atoi(argv[2]), tmp_shift;
  int inc = atoi(argv[3]);
  int out = FALSE;
  int n = 0;
  unsigned char buffer[b_size];
  unsigned char carry = 0, n_carry = 0;

  setbuf(stdout, NULL);

  while(TRUE) {

    ssize_t bytes = read(STDIN_FILENO, buffer, b_size);

    if(bytes == -1) {
      perror(”error reading stdin”);
      exit(EXIT_FAILURE);
    }

    if(inc > 0) {
      if(n == (32 * 32 * b_size)) { num_shift += 1; n = 0; };
      if(num_shift == (24 + 1)) { num_shift = 0; };

      n += bytes;
    }

    if(num_shift != 0) {

      for(int i = 0; i < ceil(num_shift / 8.0); i++) {

	if (i == (ceil(num_shift / 8.0) - 1)) { 
	  if ((num_shift % 8) != 0) {
	    tmp_shift = num_shift % 8;
	  }
	  else{          
	    tmp_shift = 8;
	  }
	  out = TRUE; 
	}
	else {
	  tmp_shift = 8;
	}

	for(int i = 0; i < b_size; i++) {
	  n_carry = buffer[i] << (8 - tmp_shift);
	  buffer[i] = (buffer[i] >> tmp_shift) | carry; 
	  carry = n_carry;

	  if (out) { putchar(buffer[i]); }; 
	}

	carry = 0;
	n_carry = 0;
	out = FALSE;
      } 
    }
    else {
      for(int i = 0; i < b_size; i++) {
	putchar(buffer[i]);
      }
    } 
  }

 return 0; 
}

filler.sh 

file=./Helvetica.ttc
fileSize=$(stat -f%z $file)
sr=44100
fr=30
vWidth=720
vHeight=720
dWidth=$vWidth
dHeight=$vHeight
numPixD=$(($dWidth * $dHeight))
smpPerFrame=$(($sr / $fr))
fillD=$(($numPixD / $smpPerFrame))
skipD=$(($fillD * $smpPerFrame))
numIter=$(($fileSize / $smpPerFrame))

for (( i = 0; i <= $numIter; i++ ))
do
  currPixMul=$(($i % $fillD))
  currPixCount=$(($smpPerFrame * ($currPixMul + 1)))
  currPixFill=$(($numPixD - $currPixCount))

  if [ $i -eq 0 ]
  then
    offset=0
  elif [ $currPixMul -eq 0 ]
  then
    offset=$(($offset + $skipD))
  fi

  tail -c +$offset $file | head -c $currPixCount >> ./frames/frame$i.raw
  head -c $currPixFill /dev/zero >> ./frames/frame$i.raw

  total=$(($currPixCount + $currPixFill))

  printf ”\nwritten frame$i\n”
done
